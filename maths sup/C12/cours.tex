\begin{proposition}{12.50 (4)}{propriétés des nombres premiers}
    Si $n > 1$ n'a pas de diviseur dans $\seg{2}{\sqrt{n}}$, alors $n$ est premier.
\end{proposition}

\begin{proposition}{12.50 (5)}{propriétés des nombres premiers}
    Si $p \in \mb{P}$, alors pour tous entiers $a$ et $b$, $(a+b)^p \equiv a^p + b^p \mod p$.
\end{proposition}
