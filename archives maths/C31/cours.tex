\begin{theoreme}{31.35}{nombre d'applications injectives entre deux ensemble finis}
Soit $E$ et $F$ deux ensembles finis. Il existe : \begin{itemize}
    \item 0 application injective de $E$ dans $F$ si $\abs{E} > \abs{F}$
    \item $\displaystyle \frac{\abs{F}!}{(\abs{F} - \abs{E})!}$ sinon.
\end{itemize}
\end{theoreme}
