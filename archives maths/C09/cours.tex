\begin{definition}{9.47 (0)}{définition topologique de densité dans $\R$}
    Soit $A$ une partie non vide de $\R$. On dit que $A$ est \notion{dense dans $\R$} lorsque tout intervalle ouvert non vide de $\R$ rencontre $A$ :
    $$\forall (x,y) \in \R^2,\, a<b \implies ]a\,; b [ \cap A \neq \varnothing$$
\end{definition}

\begin{definition}{9.47 (1)}{définition métrique de densité dans $\R$}
    Soit $A$ une partie non vide de $\R$. On dit que $A$ est \notion{dense dans $\R$} lorsque de tout réel il existe aussi proche que l'on veut un élément de $A$ :
    $$\forall x \in \R,\, \forall \epsilon > 0 ,\, \exists a \in A,\, \abs{x-a}< \epsilon$$
\end{definition}
