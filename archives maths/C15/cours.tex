\begin{proposition}{15.6}{limite en un point du domaine de définition}
    Soit $f:I \to \R$ une fonction, $x_0 \in I$. Si $f$ admet une limite finie en $a$, alors $\displaystyle \lim_{x \to x_0} = f(x_0)$.
\end{proposition}

\begin{definition}{15.66}{compacité}
    Soit $I \subset \R$. On dit que $I$ est compact si de toute suite d'éléments de $I$ on peut extraire une suite convergeant dans $I$.
\end{definition}
