\begin{proposition}{13.26}{formule de Leibniz}
    Soit $\mb{A}$ un anneau commutatif. Soit $P$ et $Q$ dans $\mb{A}[X]$ et $n \in \N$. Alors :
    $$(PQ)^n = \somme{k=0}{n}\binom{k}{n}P^{(k)}Q^{(n-k)}$$
\end{proposition}
