\begin{definition}{16.102}{ - $P$-évaluation}
    Soit $P \in \mc{I}_{\mb{K}[X]}$ et $Q \in \mb{K}[X]$ non nul. l'ensemble $\{r \in \mb{N}, P^n \mid Q\}$ est non vide (0 appartient à cet emseble), et majoré, donc admet un plus grand élément qu'on note $v_P(Q)$ et appelle \notion{$P$-évaluation} de $Q$. \\ Par convention, $v_P(0) = +\infty$
\end{definition}
