\begin{definition}{36.2}{boules ouverte et fermée}
    Soit $E$ un espace vectoriel normé. Soit $r>0$ et $a \in E$. On appelle :
    \begin{itemize}
        \item \notion{boule ouverte de rayon $r$ centrée en $a$} l'ensemble $B(a,r) = \{ x \in E,\, \norme{a-x} < r\}$
        \item \notion{boule fermée de rayon $r$ centrée en $a$} l'ensemble $\overline{B}(a,r) = \{ x \in E,\, \norme{a-x} \leq r\}$
    \end{itemize}
\end{definition}

\begin{definition}{36.3}{voisinage d'un point}
    Soit $E$ un espace vectoriel normé, $a \in E$. On appelle \notion{voisinage de $a$} toute partie $V$ de $E$ contenant une boule ouverte centrée en $a$. 
\end{definition}

\begin{definition}{36.4}{ensemble des voisinages d'un point}
    Soit $E$ un espace vectoriel normé, $a\in E$. On note $\mc{V}_E(a)$ l'ensemble des voisinage de $a$ dans $E$.
\end{definition}
