\begin{definition}{21.32}{endomorphisme nilpotent}
    Soit $E$ un espace vectoriel. On dit qu'un endomorphisme de $E$ est \notion{nilpotent} s'il existe $n \in \N^*$ tel que $u^n = 0_{\mc{L}(E)}$. L'entier $n$ est alors appelé \notion{indice de nilpotence de $u$}.
\end{definition}

\begin{definition}{21.41}{polynôme minimal}
    L'ensemble $I$ des polynômes annulateur d'un endomorphisme $u$ forme un idéal de $\mb{K}[X]$. \\
    Comme $\mb{K}[X]$ est principal, il existe un unique polynôme unitaire $D \in \mb{K}[X]$ tel que : $$ I = D \mb{K}[X] $$ Ce polynôme $D$ est appelé \notion{polynôme minimal} de $u$. Il n'existe que si $u$ admet au moins un polynôme annulateur non nul.
\end{definition} 

\begin{definition}{21.48}{endomorphisme diagonalisable, valeurs propres}
    Un endomorphisme $u$ de $E$ est \notion{diagonalisable} s'il existe une base $(b_i)_{i\in I}$ de $E$ et une famille $(\lambda_i)_{i\in I}$ de scalaires telles que : 
    $$\forall i \in I,\, u(b_i) = \lambda_i b_i$$
    les $\lambda_i$ sont appelés \notion{valeurs propres} de $u$.
\end{definition}
