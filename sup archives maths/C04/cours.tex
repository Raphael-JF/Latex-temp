\begin{proposition}{4.25 (2)}{dilation, contraction horizontale d'une fonction périodique}
    Soit $T>0$, $A$ une partie de $\R$, $f$ une fonction $T$-périodique sur $A$. Soit $\omega > 0$. La fonction $x \mapsto f(\omega x)$ est $\displaystyle \frac{T}{\omega}$-périodique sur l'ensemble $\displaystyle \frac{1}{\omega}A$.
\end{proposition}
