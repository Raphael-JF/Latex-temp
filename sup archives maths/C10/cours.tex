\begin{theoreme}{10.37}{sous-groupe de $(\mb{Z}, +)$}
    Soit $G$ un sous-groupe de $(\mb{Z}, +)$. Alors il existe $n \in \mb{Z}$ tel que $G = n \mb{Z}$.
\end{theoreme}

\begin{theoreme}{10.39}{sous-groupe de $(\R, +)$}
    Soit $G$ un sous-groupe de $(\R, +)$. Alors soit $G$ est dense dans $\R$, soit il existe $\alpha \in \R_+$ tel que $G = \alpha \mb{Z}$.
\end{theoreme}

\begin{theoreme}{10.60}{règle de calcul dans un anneau}
    Soit $(A,+,\times)$ un anneau. Si $(x,y) \in A^2$ commutent, alors les formules suivantes s'appliquent à $x$ et $y$.
    \begin{itemize}
        \item Binôme de Newton
        \item Identité de Bernoulli
    \end{itemize}

\end{theoreme}
